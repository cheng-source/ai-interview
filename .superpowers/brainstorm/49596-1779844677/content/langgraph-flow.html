State 数据结构

interface InterviewState {
  // 候选人信息 (parse_resume 产出)
  candidate: {
    name: string, skills: string[], experience: number
    projects: string[], strengths: string[], gaps: string[]
  }

  // 当前阶段标识
  currentStage: "icebreaker" | "technical" | "behavioral" | "qa" | "done"

  // 技术轮状态
  techRound: {
    currentTopic: string, currentQuestion: Question | null
    questionsAsked: Question[], depth: number
    topics: string[]   // 待覆盖的技术领域
  }

  // 行为轮状态
  behavioralRound: {
    currentQuestion: Question | null
    questionsAsked: Question[], depth: number
    competencies: string[]  // 待评估能力维度
  }

  // 全量回答历史
  answerHistory: Array<{
    stage: string, question: Question
    answer: string, evaluation: Evaluation
  }>

  // 评分累加
  scores: { technical: number, behavioral: number, overall: number }

  // 消息流 (用于推送到前端)
  messages: BaseMessage[]
}

两条关键 Conditional Edge

技术轮路由决策
function routeInTechnical(state):
  lastEval = state.answerHistory[last].evaluation

  // 答得不够深 + 追问未超过3层 → 追问
  if (lastEval.surfaceLevel && state.techRound.depth < 3):
    return "tech_follow_up"

  // 深度够了 或 追问已达上限 → 下一题
  if (state.techRound.topics.length > 0):
    return "tech_next_topic"

  // 所有知识点已覆盖 → 进入行为轮
  return "behavioral_round"
行为轮路由决策
function routeInBehavioral(state):
  lastEval = state.answerHistory[last].evaluation

  // 回答空洞 + 追问未超过2层 → 追问
  if (lastEval.vague && state.behavioralRound.depth < 2):
    return "behavioral_follow_up"

  // 清晰完整 → 下一个能力维度
  if (state.behavioralRound.competencies.length > 0):
    return "behavioral_next_question"

  // 所有能力维度已覆盖 → 反问环节
  return "candidate_qa"
