RAG Tool 定义

// LangChain Tool Schema
const companyKnowledgeTool = tool(
  async ({ query }) => {
    // 1. Embedding 查询向量化
    const embedding = await embeddings.embedQuery(query);

    // 2. PostgreSQL + pgvector 检索 top-3 最相关文档
    const docs = await db.query(`
      SELECT content, metadata
      FROM company_docs
      ORDER BY embedding <=> $1
      LIMIT 3
    `, [embedding]);

    // 3. 返回检索上下文
    return docs.map(d => d.content).join("\n---\n");
  },
  {
    name: "retrieve_company_info",
    description: "检索公司内部知识库，获取福利制度、技术栈、团队信息等",
    schema: z.object({ query: z.string() })
  }
);

LangGraph 中如何使用

// candidate_qa 节点定义
async function candidateQaNode(state: InterviewState) {
  const modelWithTools = llm.bindTools([companyKnowledgeTool]);

  const response = await modelWithTools.invoke([
    new SystemMessage(`你是公司的AI面试官，候选人正在反问环节。
      如果问题涉及公司信息（福利、文化、技术栈、团队），
      必须先调用 retrieve_company_info 检索文档再回答。
      其他通用问题直接礼貌回复。`),
    new HumanMessage(state.lastCandidateQuestion)
  ]);

  // LangGraph 自动处理 Tool Call 循环
  // 如果 LLM 决定调用 tool → 执行 → 把结果喂回 LLM → 输出答案

  return {
    messages: [...state.messages, response],
    qaCount: state.qaCount + 1
  };
}

依赖清单

组件 选型 说明
向量数据库 pgvector PostgreSQL 扩展，和面试数据同库，运维最简单
Embedding 模型 OpenAI text-embedding-3-small 便宜且够用，后续可换本地模型降成本
文档加载 @langchain/community document loaders 支持 PDF、DOCX、Markdown、纯文本
文档管理 NestJS Service + 管理界面 HR 在管理端上传/更新公司文档
